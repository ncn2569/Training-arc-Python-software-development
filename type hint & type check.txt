dynamic check thì thường kết hợp với pydantic để báo lỗi khi không đúng kiểu dữ liệu 
From Pydantic import BaseModel
# Pydantic sẽ tự động ép kiểu hoặc báo lỗi ngay lập tức nếu truyền sai dữ liệu
class NguoiDung(BaseModel):
    id: int
    ten: str
    email: str | None = None

# Đúng kiểu -> Chạy mượt mà
u1 = NguoiDung(id=1, ten="Nguyen") 

# Sai kiểu -> Pydantic ném ra lỗi ValidationError lập tức lúc runtime
u2 = NguoiDung(id="abc", ten="Nguyen")
